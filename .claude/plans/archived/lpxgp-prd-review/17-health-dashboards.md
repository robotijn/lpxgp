### 17.4 Data Model for Analytics

```sql
-- Materialized view for funnel metrics (refresh daily)
CREATE MATERIALIZED VIEW pipeline_funnel_metrics AS
SELECT
    -- Time dimensions
    DATE_TRUNC('month', oa.created_at) as month,
    DATE_TRUNC('week', oa.created_at) as week,

    -- Client dimensions
    c.id as client_id,
    c.name as client_name,
    cl.client_type,
    CASE WHEN c.type = 'lp' THEN c.lp_type ELSE NULL END as lp_type,

    -- Stage counts
    COUNT(*) FILTER (WHERE oa.stage = 'identified') as identified,
    COUNT(*) FILTER (WHERE oa.stage = 'shortlisted') as shortlisted,
    COUNT(*) FILTER (WHERE oa.stage IN ('cold_outreach_sent', 'intro_made')) as contacted,
    COUNT(*) FILTER (WHERE oa.stage LIKE 'responded%') as responded,
    COUNT(*) FILTER (WHERE oa.stage IN ('call_scheduled', 'call_completed', 'meeting_scheduled', 'meeting_completed')) as meetings,
    COUNT(*) FILTER (WHERE oa.stage LIKE 'dd_%') as due_diligence,
    COUNT(*) FILTER (WHERE oa.stage = 'committed') as committed,
    COUNT(*) FILTER (WHERE oa.stage IN ('passed_by_gp', 'passed_by_lp', 'lost_to_competitor')) as lost,

    -- Commitment amounts
    SUM(oa.commitment_amount_mm) FILTER (WHERE oa.stage = 'committed') as total_committed_mm,
    AVG(oa.commitment_amount_mm) FILTER (WHERE oa.stage = 'committed') as avg_commitment_mm,

    -- Timing metrics
    AVG(EXTRACT(EPOCH FROM (oa.first_response_at - oa.first_contact_at)) / 86400)
        FILTER (WHERE oa.first_response_at IS NOT NULL) as avg_days_to_response,

    AVG(EXTRACT(EPOCH FROM (oa.committed_at - oa.first_contact_at)) / 86400)
        FILTER (WHERE oa.committed_at IS NOT NULL) as avg_days_to_commitment

FROM outreach_activities oa
JOIN clients cl ON oa.fund_id IN (SELECT id FROM client_funds WHERE client_id = cl.id)
                OR oa.lp_id = cl.company_id
JOIN companies c ON cl.company_id = c.id
GROUP BY 1, 2, 3, 4, 5, 6;

CREATE INDEX idx_funnel_metrics_month ON pipeline_funnel_metrics(month);
CREATE INDEX idx_funnel_metrics_client ON pipeline_funnel_metrics(client_id);

-- Refresh daily
-- REFRESH MATERIALIZED VIEW pipeline_funnel_metrics;

-- Data quality metrics
CREATE MATERIALIZED VIEW data_quality_metrics AS
SELECT
    DATE_TRUNC('day', NOW()) as snapshot_date,

    -- Company counts by type
    COUNT(*) FILTER (WHERE type IN ('lp', 'both')) as lp_count,
    COUNT(*) FILTER (WHERE type IN ('gp', 'both')) as gp_count,

    -- Quality scores
    AVG(cqs.quality_score) as avg_quality_score,
    AVG(cqs.quality_score) FILTER (WHERE c.type IN ('lp', 'both')) as avg_lp_quality,
    AVG(cqs.quality_score) FILTER (WHERE c.type IN ('gp', 'both')) as avg_gp_quality,

    -- Field completeness (LP)
    AVG(CASE WHEN c.lp_type IS NOT NULL THEN 1 ELSE 0 END) FILTER (WHERE c.type IN ('lp', 'both')) as lp_type_pct,
    AVG(CASE WHEN c.aum_usd_mm IS NOT NULL THEN 1 ELSE 0 END) FILTER (WHERE c.type IN ('lp', 'both')) as aum_pct,
    AVG(CASE WHEN c.allocation_pe_pct IS NOT NULL THEN 1 ELSE 0 END) FILTER (WHERE c.type IN ('lp', 'both')) as allocation_pct,
    AVG(CASE WHEN c.check_size_min_mm IS NOT NULL THEN 1 ELSE 0 END) FILTER (WHERE c.type IN ('lp', 'both')) as check_size_pct,

    -- Field completeness (GP)
    AVG(CASE WHEN c.strategies IS NOT NULL THEN 1 ELSE 0 END) FILTER (WHERE c.type IN ('gp', 'both')) as strategies_pct,
    AVG(CASE WHEN c.flagship_fund_size_mm IS NOT NULL THEN 1 ELSE 0 END) FILTER (WHERE c.type IN ('gp', 'both')) as fund_size_pct,

    -- Freshness
    COUNT(*) FILTER (WHERE c.updated_at > NOW() - INTERVAL '30 days') as updated_30d,
    COUNT(*) FILTER (WHERE c.updated_at > NOW() - INTERVAL '90 days') as updated_90d,
    COUNT(*) FILTER (WHERE c.updated_at > NOW() - INTERVAL '365 days') as updated_365d,
    COUNT(*) FILTER (WHERE c.updated_at <= NOW() - INTERVAL '365 days') as stale_count

FROM companies c
LEFT JOIN company_quality_scores cqs ON c.id = cqs.company_id
GROUP BY 1;

-- Historical snapshots for trend
CREATE TABLE data_quality_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    snapshot_date DATE NOT NULL,
    metrics JSONB NOT NULL,
    created_at TIMESTAMP DEFAULT NOW(),
    UNIQUE(snapshot_date)
);
```

### 17.7 Chart Library Choice

**Recommendation: Chart.js via CDN**

Pros:
- Simple, declarative API
- Lightweight (~60KB)
- Good defaults, minimal config
- Responsive out of the box
- Works well with HTMX updates

```html
<!-- In base.html -->
<script src="https://cdn.jsdelivr.net/npm/chart.js"></script>

<!-- In page -->
<canvas id="funnelChart"></canvas>
<script>
  new Chart(document.getElementById('funnelChart'), {
    type: 'bar',
    data: {...},
    options: {indexAxis: 'y'}  // Horizontal for funnel effect
  });
</script>
```

Alternative for complex visualizations: D3.js (more powerful, steeper learning curve)
